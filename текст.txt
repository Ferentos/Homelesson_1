int task_2a() {
	setlocale(LC_CTYPE, "RUS");
	int mile_1;
	puts("Введите число морских миль");
	scanf("%d", &mile_1);
	printf("%d морских миль это %d километров ", mile_1, mile_1 * 1852);//Переводим морские мили в километры
	printf("%d морских миль это %d римских миль ", mile_1, mile_1 * 1852 / 1475);//Переводим морские мили в римские

	getchar();
	return 0;

}

int task_1() {
